1. Check the deployment status:
   kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "netbird-coredns.name" . }},app.kubernetes.io/instance={{ .Release.Name }}"

2. View logs:
   kubectl logs --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "netbird-coredns.name" . }},app.kubernetes.io/instance={{ .Release.Name }}"

3. Access the API (port-forward):
   export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "netbird-coredns.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
   kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:{{ .Values.config.apiPort }}
   
   Then access:
   - Health check: http://localhost:8080/health
   - Records API: http://localhost:8080/api/v1/records

4. DNS traffic is routed through the NetBird peer network.
